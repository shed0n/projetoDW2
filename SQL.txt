CREATE TABLE USUARIO (

NOME VARCHAR(50) NOT NULL,
SENHA VARCHAR(50) NOT NULL )


CREATE TABLE PATRIMONIO (
CODIGO INT PRIMARY KEY,
SITUACAO  VARCHAR(50),
RESPONSAVEL VARCHAR(50) NOT NULL,
SETOR VARCHAR(50),
LOCAL VARCHAR(50),
MATERIAL VARCHAR(50),
MARCA VARCHAR(50),
MODELO VARCHAR(50),
DATA VARCHAR(50) )

OUTROS VARCHAR(50) )
